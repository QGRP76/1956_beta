		}

		every_state = {
			limit = {
				OR = {
					state = 1048
					state = 1045
					state = 676
				}
			}
			ENG = {
				remove_resource_rights = PREV
				add_opinion_modifier = {
					target = IRQ
					modifier = large_decrease
				}
			}
		}
		remove_ideas = IRQ_oil_concessions
		hidden_effect = {
			if = {
				limit = {
					NOT={
						has_idea = IRQ_faisal_idea

				}

			}
			every_neighbor_country = {

					limit = {
						NOT = {
							tag = ENG
						}

				country_event = { id = iraq_hist.2 hours = 12 }
			}
			ITA = {
				country_event = { id = iraq_hist.2 hours = 12 }
			}
			if = {
				limit = {
					ENG = {
						controls_state = 656
						OR = {
							has_government = democratic
							has_completed_focus = ENG_the_kings_party
							has_completed_focus = ENG_home_defence
							has_completed_focus = ENG_global_defense
						}
					}

						id = iraq_hist.6
