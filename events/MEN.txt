country_event = { # The Zhangbei Meeting
	id = men.10
	title = men.10.t
	desc = men.10.d
	
	picture = GFX_report_event_chinese_japanese_handshake
	
	fire_only_once = yes

	is_triggered_only = yes
	
	option = {
		name = men.10.a
		if = {
			limit = {
				has_country_flag = MEN_japanese_gambit
			}
			add_political_power = 15
			add_stability = 0.03
			custom_effect_tooltip = MEN_unforseen_consequences
			hidden_effect = {
				country_event = { id = men.12 hours = 12 }
			}
		}
		else = {
			add_popularity = {
				ideology = neutrality
				popularity = 0.05
			}
			add_stability = 0.10
			effect_tooltip = {
				unlock_national_focus = MEN_petition_nima-odsor_to_stay_in_shangbei
			}
		}
	}

	option = {
		name = men.10.b
		trigger = {
			has_country_flag = MEN_japanese_gambit
		}
		add_political_power = -35
		add_popularity = {
			ideology = neutrality
			popularity = -0.05
		}
		add_stability = -0.10
		custom_effect_tooltip = MEN_prevent_destabilization
		hidden_effect = {
			country_event = { id = men.11 hours = 12 }
		}
	}
}
